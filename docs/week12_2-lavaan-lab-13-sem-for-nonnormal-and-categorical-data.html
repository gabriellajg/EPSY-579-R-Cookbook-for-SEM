Write out syntax for a one-factor CFA model:

cfaSyn <- "
    odd =~ odd1 + odd2 + odd3 + odd4 + odd5 + odd6 + odd7 + odd8
"

Fit the one-factor model:

mlrFit <- lavaan::sem(cfaSyn, 
              data = nnorm_dat, 
              fixed.x = FALSE,
              estimator = "mlr")
